# 元家乡 2050 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 在 48 小时内交付一个可演示、可扫码体验的 Web demo——访客通过地图与"村管家龙寻"对话获得定制旅居方案，与小梅庄主、黄喆（云村民）、浅予 三个主理人分身深聊，最终在线下单。所有 AI 回答可追溯到 wiki markdown 源文件。

**Architecture:** Vault（markdown + persona-cards JSON）为单一真知库；Node Express 后端通过 PersonaLoader 运行时拼装 system prompt；前端 Vite + React 渲染地图（实/云双类点位）+ 对话面板 + 积分系统。LLM 通过 50 行手写 wrapper 支持 Claude/DeepSeek/OpenAI 切换。村管家用 L0-L3 分层蒸馏处理《龙潭造梦记》390 页扫描书，避免引入向量库。

**Tech Stack:**
- Frontend: Vite 5 + React 18 + Tailwind 3 + Zustand + React Router 6
- Backend: Node 20 + Express 4 + @anthropic-ai/sdk + openai (兼容 DeepSeek)
- 数据: 文件系统 markdown + gray-matter parser + JSON
- 测试: Vitest（关键模块），其余靠端到端手测
- 部署: Vercel (前端) + Render/Railway (后端)
- 蒸馏: 自写 scripts (Node)，PaddleOCR 处理 PDF

**Related specs:**
- `docs/superpowers/specs/2026-05-20-zhuliren-fenshen-business.md`
- `docs/superpowers/specs/2026-05-20-zhuliren-fenshen-architecture.md`

```js
// backend/services/llm.js
import Anthropic from '@anthropic-ai/sdk';
import OpenAI from 'openai';

const PROVIDER = process.env.LLM_PROVIDER || 'claude';
const MODEL = process.env.LLM_MODEL;
const API_KEY = process.env.LLM_API_KEY;
const BASE_URL = process.env.LLM_BASE_URL;
const CACHE_ENABLED = process.env.PROMPT_CACHE_ENABLED === 'true';

let _client;

function getClient() {
  if (_client) return _client;
  if (PROVIDER === 'claude') {
    _client = new Anthropic({ apiKey: API_KEY });
  } else if (PROVIDER === 'deepseek') {
    _client = new OpenAI({
      apiKey: API_KEY,
      baseURL: BASE_URL || 'https://api.deepseek.com/v1'
    });
  } else if (PROVIDER === 'openai') {
    _client = new OpenAI({ apiKey: API_KEY });
  } else {
    throw new Error(`Unknown LLM_PROVIDER: ${PROVIDER}`);
  }
  return _client;
}

/**
 * 统一 chat 接口
 * @param {object} opts
 * @param {string} opts.system - system prompt
 * @param {Array<{role,content}>} opts.messages - 对话历史
 * @param {Array} [opts.tools] - tool 定义 (provider-specific shape)
 * @param {string|object} [opts.toolChoice] - 强制 tool 使用
 * @param {number} [opts.maxTokens=1024]
 * @returns {Promise<{text: string, toolUse: object|null, usage: object}>}
 */
export async function chat({ system, messages, tools, toolChoice, maxTokens = 1024 }) {
  const client = getClient();

  if (PROVIDER === 'claude') {
    return chatClaude(client, { system, messages, tools, toolChoice, maxTokens });
  }
  // deepseek / openai 共用 OpenAI 协议
  return chatOpenAICompat(client, { system, messages, tools, toolChoice, maxTokens });
}

async function chatClaude(client, { system, messages, tools, toolChoice, maxTokens }) {
  const req = {
    model: MODEL,
    max_tokens: maxTokens,
    system: CACHE_ENABLED
      ? [{ type: 'text', text: system, cache_control: { type: 'ephemeral' } }]
      : system,
    messages
  };
  if (tools) {
    req.tools = tools;
    if (toolChoice) req.tool_choice = toolChoice;
  }
  const resp = await client.messages.create(req);

  let text = '';
  let toolUse = null;
  for (const block of resp.content) {
    if (block.type === 'text') text += block.text;
    if (block.type === 'tool_use') toolUse = { name: block.name, input: block.input };
  }
  return { text, toolUse, usage: resp.usage };
}

async function chatOpenAICompat(client, { system, messages, tools, toolChoice, maxTokens }) {
  const req = {
    model: MODEL,
    max_tokens: maxTokens,
    messages: [
      { role: 'system', content: system },
      ...messages
    ]
  };
  if (tools) {
    // OpenAI tool 定义 shape 不同于 Claude
    req.tools = tools.map(t => ({
      type: 'function',
      function: { name: t.name, description: t.description, parameters: t.input_schema }
    }));
    if (toolChoice) {
      req.tool_choice = typeof toolChoice === 'object'
        ? { type: 'function', function: { name: toolChoice.name } }
        : toolChoice;
    }
  }
  const resp = await client.chat.completions.create(req);
  const msg = resp.choices[0].message;
  let toolUse = null;
  if (msg.tool_calls?.length) {
    const call = msg.tool_calls[0];
    toolUse = { name: call.function.name, input: JSON.parse(call.function.arguments) };
  }
  return { text: msg.content || '', toolUse, usage: resp.usage };
}

export function getProviderInfo() {
  return { provider: PROVIDER, model: MODEL, cacheEnabled: CACHE_ENABLED };
}
```

```js
// backend/services/activitiesLoader.js
import fs from 'fs/promises';
import path from 'path';
import matter from 'gray-matter';

const VAULT = process.env.VAULT_PATH || '../vault';
const ACTIVITIES_ROOT = path.resolve(VAULT, 'wiki/activities');

let _cache = new Map();
const TTL_MS = 60 * 60 * 1000; // 1h, 活动状态切换慢

function classifyStatus(activity, todayStr) {
  if (activity.status_override) return activity.status_override;
  if (activity.date_end) {
    if (todayStr > activity.date_end) return 'past';
    if (todayStr < activity.date_start) return 'upcoming';
    return 'ongoing';
  }
  if (activity.date < todayStr) return 'past';
  if (activity.date > todayStr) return 'upcoming';
  return 'ongoing';
}

function todayString() {
  const d = new Date();
  return `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}-${String(d.getDate()).padStart(2, '0')}`;
}

export async function loadActivitiesForSpace(spaceSlug) {
  const cacheKey = `${spaceSlug}:${todayString()}`;
  const cached = _cache.get(cacheKey);
  if (cached && Date.now() - cached.at < TTL_MS) return cached.data;

  const dir = path.join(ACTIVITIES_ROOT, spaceSlug);
  let files;
  try {
    files = await fs.readdir(dir);
  } catch (e) {
    if (e.code === 'ENOENT') return { featured: [], archive: [] };
    throw e;
  }

  const today = todayString();
  const items = await Promise.all(
    files.filter(f => f.endsWith('.md')).map(async f => {
      const raw = await fs.readFile(path.join(dir, f), 'utf8');
      const { data, content } = matter(raw);
      return { ...data, body: content, status: classifyStatus(data, today) };
    })
  );

  const featured = items
    .filter(a => ['upcoming', 'ongoing'].includes(a.status))
    .sort((a, b) => (a.date || a.date_start).localeCompare(b.date || b.date_start))
    .slice(0, 5);
  const archive = items
    .filter(a => a.status === 'past')
    .sort((a, b) => (b.date || b.date_end).localeCompare(a.date || a.date_end));

  const data = { featured, archive };
  _cache.set(cacheKey, { data, at: Date.now() });
  return data;
}

export function invalidateActivitiesCache() {
  _cache.clear();
}
```

```js
// backend/services/personaLoader.js
import fs from 'fs/promises';
import path from 'path';
import matter from 'gray-matter';
import { loadActivitiesForSpace } from './activitiesLoader.js';
import { getSpaceBySlug } from './spacesLoader.js';

const VAULT = process.env.VAULT_PATH || '../vault';
const CARDS_DIR = path.resolve(VAULT, 'persona-cards');
const WIKI_DIR = path.resolve(VAULT, 'wiki');

const _cache = new Map();
const TTL_MS = 5 * 60 * 1000;

export async function loadPersona(slug) {
  const cached = _cache.get(slug);
  if (cached && Date.now() - cached.at < TTL_MS) return cached.data;

  const cardPath = path.join(CARDS_DIR, `${slug}.json`);
  let card;
  try {
    card = JSON.parse(await fs.readFile(cardPath, 'utf8'));
  } catch (e) {
    throw new Error(`Persona card not found: ${cardPath}`);
  }

  // 解析 source_refs 拉原文
  const sources = await loadSources(card.x_xianjian_source_refs || []);

  // 拉关联空间的活动
  const spaceSlug = card.x_xianjian_space_slug;
  let activities = { featured: [], archive: [] };
  let space = null;
  if (spaceSlug) {
    space = await getSpaceBySlug(spaceSlug);
    activities = await loadActivitiesForSpace(spaceSlug);
  }

  const systemPrompt = assembleSystemPrompt({ card, sources, activities, space });

  const data = { card, systemPrompt, space, activities };
  _cache.set(slug, { data, at: Date.now() });
  return data;
}

async function loadSources(refs) {
  return Promise.all(refs.map(async ref => {
    const fullPath = path.join(VAULT, ref);
    const raw = await fs.readFile(fullPath, 'utf8');
    const { data, content } = matter(raw);
    return { ...data, body: content, ref };
  }));
}

function assembleSystemPrompt({ card, sources, activities, space }) {
  const parts = [];

  // L0 灵魂层
  parts.push(card.system_prompt || '');
  parts.push('');
  parts.push('## 人设描述', card.description || '');
  parts.push('## 性格', card.personality || '');
  parts.push('## 场景', card.scenario || '');
  if (card.mes_example) parts.push('## 示范对话', card.mes_example);

  // Evidence ledger
  if (card.x_xianjian_evidence_ledger?.length) {
    parts.push('## Evidence Ledger');
    for (const entry of card.x_xianjian_evidence_ledger) {
      parts.push(`### [${entry.claim_id}] ${entry.claim}`);
      parts.push(`置信度: ${entry.confidence}`);
      if (entry.abstain_if_asked_beyond?.length) {
        parts.push(`超出范围话题 (需 abstain): ${entry.abstain_if_asked_beyond.join('; ')}`);
      }
      for (const s of entry.sources || []) {
        parts.push(`- [${s.source_id}] "${s.excerpt}"`);
      }
    }
  }

  // Sources 全文
  parts.push('## <sources>');
  for (const src of sources) {
    parts.push(`### <source id="${src.id}" date="${src.date}" tags="${(src.tags || []).join(',')}">`);
    parts.push(src.body);
    parts.push('### </source>');
  }
  parts.push('## </sources>');

  // 当前空间 + 活动
  if (space) {
    parts.push('## 我的空间');
    parts.push(`${space.name} (${space.space_kind === 'cloud' ? '云空间' : '在地实空间'})`);
    parts.push(space.short_pitch || '');
  }
  if (activities.featured?.length) {
    parts.push('## <featured_activities>');
    for (const a of activities.featured) {
      parts.push(`- ${a.title} | ${a.date || a.date_start} | ${a.price || ''}`);
      if (a.body) parts.push(a.body.split('\n').slice(0, 3).join(' '));
    }
    parts.push('## </featured_activities>');
  }
  if (activities.archive?.length) {
    parts.push('## <activity_archive>');
    for (const a of activities.archive.slice(0, 10)) {
      parts.push(`- ${a.title} | ${a.date || a.date_end}`);
    }
    parts.push('## </activity_archive>');
  }

  return parts.join('\n');
}

export function invalidatePersonaCache(slug) {
  if (slug) _cache.delete(slug);
  else _cache.clear();
}
```

```js
// frontend/src/lib/yuanxp.js
import { create } from 'zustand';

const REWARDS = {
  talk_steward_deep: 10,
  enter_space: 5,
  chat_persona_long: 20,
  browse_activity_or_work: 5,
  mock_order_complete: 30
};

export const useYuanXP = create((set, get) => ({
  xp: 0,
  events: [],
  seen: new Set(), // 去重 key: `${action}:${target}`

  award(action, target = '') {
    const key = `${action}:${target}`;
    if (get().seen.has(key)) return;
    const delta = REWARDS[action] || 0;
    if (!delta) return;
    set(state => ({
      xp: state.xp + delta,
      events: [
        { at: Date.now(), action, target, delta },
        ...state.events
      ].slice(0, 20),
      seen: new Set([...state.seen, key])
    }));
  },

  reset() {
    set({ xp: 0, events: [], seen: new Set() });
  }
}));
```

```jsx
// frontend/src/components/StewardChatPanel.jsx
import { useState } from 'react';
import { api } from '../lib/api.js';
import ItineraryCard from './ItineraryCard.jsx';
import { useYuanXP } from '../lib/yuanxp.js';

export default function StewardChatPanel({ onRecommend }) {
  const [open, setOpen] = useState(true);
  const [history, setHistory] = useState([
    { role: 'assistant', content: '你好，我是龙寻，龙潭村的数字村管家。第一次来吗？想听点啥？' }
  ]);
  const [draft, setDraft] = useState('');
  const [busy, setBusy] = useState(false);
  const award = useYuanXP(s => s.award);

  async function send() {
    if (!draft.trim() || busy) return;
    const userMsg = { role: 'user', content: draft };
    const newHistory = [...history, userMsg];
    setHistory(newHistory);
    setDraft('');
    setBusy(true);
    try {
      const resp = await api.chat({
        agent: 'longxun',
        message: draft,
        history: newHistory.slice(0, -1) // exclude current
      });
      const assistantTurn = {
        role: 'assistant',
        content: resp.text,
        toolUse: resp.toolUse
      };
      setHistory([...newHistory, assistantTurn]);
      if (resp.toolUse?.name === 'propose_itinerary') {
        const out = resp.toolUse.input;
        onRecommend?.(out.recommended_spaces || []);
        award('talk_steward_deep', 'longxun');
      }
    } catch (e) {
      setHistory([...newHistory, { role: 'assistant', content: `(出错了: ${e.message}, 切换 LLM 重试)` }]);
    } finally {
      setBusy(false);
    }
  }

  return (
    <div className={`fixed right-4 bottom-4 w-[380px] bg-white shadow-2xl rounded-lg border ${open ? '' : 'h-12 overflow-hidden'}`}>
      <button
        onClick={() => setOpen(!open)}
        className="w-full bg-stone-800 text-white px-3 py-2 flex justify-between items-center text-sm rounded-t-lg"
      >
        <span>🐲 村管家龙寻</span>
        <span>{open ? '－' : '+'}</span>
      </button>

      {open && (
        <>
          <div className="h-72 overflow-y-auto p-3 space-y-2 text-sm bg-stone-50">
            {history.map((m, i) => (
              <div key={i} className={m.role === 'user' ? 'text-right' : 'text-left'}>
                <div className={`inline-block px-3 py-1.5 rounded-lg max-w-[85%] ${
                  m.role === 'user' ? 'bg-amber-200' : 'bg-white border'
                }`}>
                  {m.content || '...'}
                </div>
                {m.toolUse?.name === 'propose_itinerary' && (
                  <ItineraryCard {...m.toolUse.input} />
                )}
              </div>
            ))}
            {busy && <div className="text-stone-400 text-xs">龙寻正在思考...</div>}
          </div>

          <div className="p-2 border-t flex gap-2">
            <input
              value={draft}
              onChange={e => setDraft(e.target.value)}
              onKeyDown={e => e.key === 'Enter' && send()}
              placeholder="说说您想要什么..."
              className="flex-1 border rounded px-2 py-1 text-sm"
              disabled={busy}
            />
            <button
              onClick={send}
              disabled={busy}
              className="bg-stone-800 text-white px-3 py-1 rounded text-sm disabled:opacity-50"
            >发送</button>
          </div>
        </>
      )}
    </div>
  );
}
```

```jsx
// frontend/src/components/PersonaChat.jsx
import { useState, useRef } from 'react';
import { api } from '../lib/api.js';
import { useYuanXP } from '../lib/yuanxp.js';

export default function PersonaChat({ personaSlug, personaName, isPlaceholder }) {
  const [history, setHistory] = useState([
    { role: 'assistant', content: `你好，我是${personaName}的数字分身。${isPlaceholder ? '（早期版本，仅能回答基础问题）' : ''}` }
  ]);
  const [draft, setDraft] = useState('');
  const [busy, setBusy] = useState(false);
  const turns = useRef(0);
  const award = useYuanXP(s => s.award);

  async function send() {
    if (!draft.trim() || busy) return;
    const userMsg = { role: 'user', content: draft };
    const newHistory = [...history, userMsg];
    setHistory(newHistory);
    setDraft('');
    setBusy(true);
    turns.current += 1;
    if (turns.current >= 5) award('chat_persona_long', personaSlug);
    try {
      const resp = await api.chat({
        agent: personaSlug,
        message: draft,
        history: newHistory.slice(0, -1)
      });
      setHistory([...newHistory, { role: 'assistant', content: resp.text }]);
    } catch (e) {
      setHistory([...newHistory, { role: 'assistant', content: `(出错了: ${e.message})` }]);
    } finally {
      setBusy(false);
    }
  }

  return (
    <div className="bg-white rounded-lg shadow border h-96 flex flex-col">
      <div className="bg-stone-100 px-3 py-2 font-medium text-sm">与 {personaName} 对话</div>
      <div className="flex-1 overflow-y-auto p-3 space-y-2 text-sm">
        {history.map((m, i) => (
          <div key={i} className={m.role === 'user' ? 'text-right' : 'text-left'}>
            <div className={`inline-block px-3 py-1.5 rounded-lg max-w-[85%] ${
              m.role === 'user' ? 'bg-amber-200' : 'bg-stone-100'
            }`}>{m.content}</div>
          </div>
        ))}
        {busy && <div className="text-stone-400 text-xs">{personaName}正在思考...</div>}
      </div>
      <div className="p-2 border-t flex gap-2">
        <input
          value={draft}
          onChange={e => setDraft(e.target.value)}
          onKeyDown={e => e.key === 'Enter' && send()}
          placeholder="说点什么..."
          className="flex-1 border rounded px-2 py-1 text-sm"
          disabled={busy}
        />
        <button onClick={send} disabled={busy} className="bg-stone-800 text-white px-3 py-1 rounded text-sm disabled:opacity-50">发送</button>
      </div>
    </div>
  );
}
```

// vault/scripts/distill.js
import 'dotenv/config';
import fs from 'fs/promises';
import path from 'path';
import matter from 'gray-matter';
import { chat } from './llm.js';

const VAULT_ROOT = path.resolve(process.cwd(), '..');
const ENTITIES_DIR = path.join(VAULT_ROOT, 'wiki/entities');
const CARDS_DIR = path.join(VAULT_ROOT, 'persona-cards');

const DISTILL_SYSTEM = `你是数字分身蒸馏专家。基于给定的人物骨架和源材料, 输出 CCv3 (Character Card V3) 格式的 JSON。

蒸馏方法论 (硬约束):

1. **形容词禁令**: 不准写"她很佛系"、"他有审美追求"。必须写成可观察行为或可执行规则。
   错: "她对消费主义客人反感"
   对: "遇到要求'五星级标准服务'的客人, 倾向于拒绝接待或推荐到隔壁"

2. **证据阈值**: 每条 evidence_ledger 的 claim 必须挂 source 引用。
   - ≥2 个源 → confidence: high
   - 1 个源 → confidence: medium, 回答时开头加"我记得..."
   - 0 个源 → 不要写, 跳过

3. **真实对白优先**: mes_example 的对话必须从源材料原话改编, 不许凭空发明该人物没说过的话。

4. **优先级机器** (dot-skill L0): system_prompt 顶部按以下优先级:
   L0 红线 (基于 sources / ledger 回答, abstain 兜底, 标 [c001] 引用) >
   L5 边界禁忌 >
   一般行为模式 >
   风格细节

5. **三语态切换**: system_prompt 末尾包含:
   - <NARRATOR>: 讲故事/价值观时
   - <OPERATOR>: 答业务/价格/活动时
   - <VALIDATOR>: 内部自检 ledger 是否有支撑

请输出严格 JSON, 不要任何 markdown 围栏或解释文字。`;

async function distill(slug) {
  const entityPath = path.join(ENTITIES_DIR, `${slug}.md`);
  const raw = await fs.readFile(entityPath, 'utf8');
  const { data: meta, content: entityBody } = matter(raw);
  const isPlaceholder = !!meta.placeholder;
  const personKind = meta.person_kind || 'host';

  // 解析 [[wikilinks]] 收集 sources
  const refs = [...entityBody.matchAll(/\[\[wiki\/([^\]]+)\]\]/g)]
    .map(m => m[1])
    .filter(p => p.startsWith('sources/') || p.startsWith('works/'));
  const sources = await Promise.all(refs.map(async ref => {
    try {
      return { ref, body: await fs.readFile(path.join(VAULT_ROOT, 'wiki', ref + '.md'), 'utf8') };
    } catch { return null; }
  })).then(arr => arr.filter(Boolean));

  const userPrompt = `
## 实体骨架 (slug: ${slug}, person_kind: ${personKind})

${entityBody}

## 源材料

${sources.map(s => `<source ref="${s.ref}">\n${s.body}\n</source>`).join('\n\n')}

---

请输出 JSON, 包含以下字段:

${isPlaceholder
  ? `占位模式: 仅生成 description (100 字内), first_mes (50 字内), mes_example (1 段);
     evidence_ledger 留空数组 [];
     system_prompt 用简单 fallback: "我是${meta.name}的早期版本数字分身, 仅能回答基础问题. 完整深聊版本下一阶段上线."
     x_xianjian_person_kind: "${personKind}"
     x_xianjian_space_slug: (从 entity 的 ## 关联空间 段落推断)
     x_xianjian_source_refs: ${JSON.stringify(refs.map(r => 'wiki/' + r + '.md'))}`
  : `完整蒸馏:
     - name, description (200字内), personality (200字内), scenario, first_mes, mes_example (3-5段)
     - system_prompt (含 L0 硬约束 + 三语态切换 + 形容词禁令)
     - x_xianjian_skills (结构化 array)
     - x_xianjian_aesthetic_keywords (5-8 个)
     - x_xianjian_evidence_ledger (至少 4 条 claim, 每条挂 source 引用)
     - x_xianjian_person_kind: "${personKind}"
     - x_xianjian_space_slug: (从 entity 的 ## 关联空间 段落推断)
     - x_xianjian_source_refs: ${JSON.stringify(refs.map(r => 'wiki/' + r + '.md'))}`
}

spec, spec_version 字段固定: "chara_card_v3", "3.0"
`;

  console.log(`[distill] ${slug}: calling LLM...`);
  const { text } = await chat({
    system: DISTILL_SYSTEM,
    messages: [{ role: 'user', content: userPrompt }],
    maxTokens: 8192
  });

  // 提取 JSON (可能 LLM 包了 ```json 围栏)
  let jsonText = text.trim();
  const fenceMatch = jsonText.match(/^```(?:json)?\s*([\s\S]*?)\s*```$/);
  if (fenceMatch) jsonText = fenceMatch[1].trim();

  let card;
  try {
    card = JSON.parse(jsonText);
  } catch (e) {
    console.error('[distill] JSON parse failed. Raw output:\n', text);
    throw e;
  }

  // 归档旧版本
  const outPath = path.join(CARDS_DIR, `${slug}.json`);
  try {
    const prev = await fs.readFile(outPath, 'utf8');
    const archivePath = path.join(CARDS_DIR, '_archive', `${slug}-${Date.now()}.json`);
    await fs.mkdir(path.dirname(archivePath), { recursive: true });
    await fs.writeFile(archivePath, prev);
    console.log(`[distill] archived old version: ${archivePath}`);
  } catch {}

  card.x_xianjian_meta = {
    distilled_at: new Date().toISOString(),
    distill_model: process.env.LLM_MODEL,
    distill_version: 'v0.1',
    placeholder: isPlaceholder
  };

  await fs.writeFile(outPath, JSON.stringify(card, null, 2));
  console.log(`[distill] ✓ ${slug} → ${outPath}`);
}

const slug = process.argv[2];

```js
// vault/scripts/distill-chapters.js
import 'dotenv/config';
import fs from 'fs/promises';
import path from 'path';
import matter from 'gray-matter';
import { chat } from './llm.js';

const VAULT_ROOT = path.resolve(process.cwd(), '..');
const SOURCES_DIR = path.join(VAULT_ROOT, 'wiki/sources');
const OUT_PATH = path.join(VAULT_ROOT, 'persona-cards/longxun.l2-chapters.json');

const SYSTEM = `你是村史摘要专家。基于给定的《龙潭造梦记》章节原文, 输出 300-500 字的中文摘要。

摘要必须包含:
- 章节涉及的时间段
- 主要人物
- 主要事件
- 涉及的空间或村庄地点
- 一句话核心要点

输出纯文本摘要, 不要 markdown 格式。`;

async function summarizeChapter(filename) {
  const raw = await fs.readFile(path.join(SOURCES_DIR, filename), 'utf8');
  const { data, content } = matter(raw);
  if (data.source_type !== 'book-chapter') return null;

  const { text } = await chat({
    system: SYSTEM,
    messages: [{ role: 'user', content: `章节标题: ${data.chapter_title}\n\n${content}` }],
    maxTokens: 1024
  });

  return {
    chapter_index: data.chapter_index,
    chapter_title: data.chapter_title,
    source_id: data.id,
    summary: text.trim()
  };
}

async function main() {
  const files = (await fs.readdir(SOURCES_DIR))
    .filter(f => f.startsWith('longtang-zaomeng-ji-ch-'))
    .sort();
  console.log(`[chapters] found ${files.length} chapter files`);

  const items = [];
  for (const f of files) {
    process.stdout.write(`  - ${f}... `);
    try {
      const item = await summarizeChapter(f);
      if (item) {
        items.push(item);
        process.stdout.write(`✓\n`);
      }
    } catch (e) {
      process.stdout.write(`× ${e.message}\n`);
    }
  }

  await fs.writeFile(OUT_PATH, JSON.stringify({
    book_title: '龙潭造梦记',
    distilled_at: new Date().toISOString(),
    chapter_summaries: items
  }, null, 2));
  console.log(`[chapters] ✓ wrote ${items.length} summaries → ${OUT_PATH}`);
}

main().catch(e => { console.error(e); process.exit(1); });
```

```js
// backend/services/personaLoader.js (extend assembleSystemPrompt)

import { loadAllSpaces } from './spacesLoader.js';
import fs from 'fs/promises';

async function assembleSystemPrompt({ card, sources, activities, space }) {
  const parts = [];

  // ... 原有 L0 + sources + activities ...

  // steward 额外: 注入聚合数据
  if (card.x_xianjian_aggregates) {
    const agg = card.x_xianjian_aggregates;

    if (agg.load_all_spaces) {
      const all = await loadAllSpaces();
      parts.push('## <village_spaces>');
      for (const s of all) {
        parts.push(`<space slug="${s.slug}" kind="${s.space_kind}" host="${s.host_person_slug}">`);
        parts.push(`  ${s.short_pitch || ''}`);
        parts.push(`  style: ${(s.style_keywords || []).join(', ')}`);
        parts.push('</space>');
      }
      parts.push('## </village_spaces>');
    }

    if (agg.load_l1_facts) {
      try {
        const l1 = JSON.parse(await fs.readFile(`${process.env.VAULT_PATH || '../vault'}/persona-cards/longxun.l1-facts.json`, 'utf8'));
        parts.push('## <village_facts>');
        for (const f of l1.facts) {
          parts.push(`[${f.fact_id}] ${f.claim} (year: ${f.year || 'n/a'}, sources: ${f.sources?.map(s => s.source_id).join(',')})`);
        }
        parts.push('## </village_facts>');
      } catch (e) {
        console.warn('[steward] l1 facts not loaded:', e.message);
      }
    }

    if (agg.load_l2_chapters) {
      try {
        const l2 = JSON.parse(await fs.readFile(`${process.env.VAULT_PATH || '../vault'}/persona-cards/longxun.l2-chapters.json`, 'utf8'));
        parts.push('## <book_chapters> (《龙潭造梦记》章节摘要,深聊村史可引用)');
        for (const c of l2.chapter_summaries) {
          parts.push(`### ch${c.chapter_index} - ${c.chapter_title}`);
          parts.push(c.summary);
        }
        parts.push('## </book_chapters>');
      } catch (e) {
        console.warn('[steward] l2 chapters not loaded:', e.message);
      }
    }
  }

  return parts.join('\n');
}
```

```jsx
      ) : (
        matched.map(c => (
          <div key={c.claim_id} className="mb-3 border-b pb-2">

            <div className="text-stone-500 mt-1">置信度: {c.confidence}</div>

              </div>

    </div>
  );
}
```
